Drafts and meta pages (home/about/blog with meta=true) are filtered out. */ -}} {{- $isTerm := eq .Kind "term" -}} {{- $termPlural := "" -}} {{- $termName := "" -}} {{- if $isTerm -}} {{- $termPlural = .Data.Plural -}} {{- $termName = .Title -}} {{- end -}} {{- $articles := slice -}} {{- range .Pages -}} {{- if and (not .Draft) (not .Params.meta) -}} {{- $articles = $articles | append . -}} {{- end -}} {{- end -}} {{- $articles = sort $articles "Date" "desc" -}} {{- $count := len $articles -}} {{- $header := "[ MANIFEST ]" -}} {{- if $isTerm -}} {{- if eq $termPlural "categories" -}} {{- $header = printf "[ MANIFEST · %s ]" (upper $termName) -}} {{- else -}} {{- $header = printf "[ MANIFEST · #%s ]" (upper $termName) -}} {{- end -}} {{- end -}} {{- /* Year range + per-year counts. The year marker rows need a count; precompute once rather than re-scanning inside the entry loop. */ -}} {{- $years := slice -}} {{- $yearCounts := dict -}} {{- range $articles -}} {{- $y := .Date.Format "2006" -}} {{- $years = $years | append $y -}} {{- $yearCounts = merge $yearCounts (dict $y (add (default 0 (index $yearCounts $y)) 1)) -}} {{- end -}} {{- $uniqYears := uniq $years -}} {{- $multiYear := gt (len $uniqYears) 1 -}} {{- /* Meta line under the header. Empty case shipped via the empty-state block. */ -}} {{- $meta := "" -}} {{- if eq $count 0 -}} {{- $meta = "" -}} {{- else if $isTerm -}} {{- $word := cond (eq $count 1) "entry" "entries" -}} {{- if eq $termPlural "categories" -}} {{- $meta = printf "// %d %s · in %s" $count $word (upper $termName) -}} {{- else -}} {{- $meta = printf "// %d %s · tagged #%s" $count $word (upper $termName) -}} {{- end -}} {{- else if $multiYear -}} {{- $yearHigh := (index $articles 0).Date.Format "2006" -}} {{- $yearLow := (index $articles (sub $count 1)).Date.Format "2006" -}} {{- $meta = printf "// %d entries · spanning %s / %s" $count $yearLow $yearHigh -}} {{- else -}} {{- $word := cond (eq $count 1) "entry" "entries" -}} {{- $meta = printf "// %d %s" $count $word -}} {{- end -}}
